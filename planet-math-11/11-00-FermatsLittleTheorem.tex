					\lsection{Fermat's little theorem}
					
					In this section, let $p$ be a natural number. 
					
					Let $a$ be an integer.
					
					Let \df{natural number} stand for \Nat.
					
					Let \df{integer} stand for \Int.
					
					\deflabel{divides}
					We say that $d$ \df{divides} $m$ iff $d\ne 0$ and there exists $r$
					such that $m= d\*r$.
				\end{definition}
				
				We write $\natdiv{d}{m}$ iff $d$ divides $m$.
				
				We say that $d$ is a \df{divisor of} $m$ iff $d$ divides $m$.
				
				\deflabel{prime}
				Assume that $p$ is a natural number greater than $1$.  Then we say that $p$
				is a \df{prime} iff each divisor of $p$ is equal to $1$ or is equal to
				$p$.
			\end{definition}
			
			\deflabel{intpownat}
			Assume that $b$ is an integer and $n$ is a natural number.  Then we define
%			Let $b$ be an integer and $n$ be a natural number.  Then we define
			$\intpownat{b}{n}\assign$
			\par$\match\ n$ with
			\begin{envMatch}
				\firstmatchitem $0$ &$\assign$& $1$
				\matchitem $m + 1$ &$\assign$& $b \* \intpownat{b}{m}$\texstop
			\end{envMatch}\cnlstop
			This exists by recursion.
		\end{definition}
		
		
		\begin{theorem}[Fermat's little theorem]
			If $p$ is prime, then for any integer $a$, such that $p$ divides $\intpownat{a}{p} - a$.
		\end{theorem}
